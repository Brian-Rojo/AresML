CODIGO REMOVIDO de Autograd.hpp:

// PROPAGATE to BASE tensor if this is a view
if (current->base && current->base->grad && current->base->grad->data && 
    current->grad && current->grad->data && 
    current->base != current) {
    float* g_base = current->base->grad->data.get() + current->base->grad->offset;
    const float* g_curr = current->grad->data.get() + current->grad->offset;
    size_t n = current->base->grad->shape.size();
    for (size_t i = 0; i < n; ++i) {
        g_base[i] += g_curr[i];
    }
}

RAZON:
Este hack hacia propagacion fuera del grafo de operaciones,
violando el principio de "cada operation tiene su backward()".
